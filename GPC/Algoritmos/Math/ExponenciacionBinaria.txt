#include<bits/stdc++.h>
#include <cstdint>
using namespace std;
#define int long long 
const long long mod =1000000007;
long long binpow(long long a, long long b) {
    a %= mod;
    long long res = 1;
    while (b > 0) {
        if (b & 1)
            res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}
